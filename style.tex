  \begin{tcolorbox}[
    enhanced,

    colframe=gray!60!black,
    boxrule=0pt,
    leftrule=2pt,
    arc=0pt,

    before skip=0.5em,
    after skip=0.5em
  ]
  \textbf{证明：}\par\vspace{2pt}

  \end{tcolorbox}
}
