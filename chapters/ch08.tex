\chapter{拉普拉斯变换\index{拉普拉斯变换}}

\begin{wulibj}[本章导读]
拉普拉斯变换\index{拉普拉斯变换}是处理常微分方程初值问题的有力工具。与傅里叶变换\index{傅里叶变换}不同，它对函数的增长性要求更宽松，且自动包含初始条件。本章我们将学习拉普拉斯变换\index{拉普拉斯变换}的定义、性质及其在求解微分方程中的应用。
\end{wulibj}

% ========== 第一节 ==========
\section{拉普拉斯变换\index{拉普拉斯变换}的定义}

\subsection{从傅里叶变换\index{傅里叶变换}到拉普拉斯变换\index{拉普拉斯变换}}

傅里叶变换\index{傅里叶变换}要求函数在整个实轴上绝对可积，很多常见函数（如 $\mathrm{e}^x$、多项式等）不满足此条件。

拉普拉斯变换\index{拉普拉斯变换}的思想是：引入衰减因子 $\mathrm{e}^{-\sigma t}$，使 $\mathrm{e}^{-\sigma t}f(t)$ 绝对可积。

\begin{dingliyi}{}{拉普拉斯变换}
设 $f(t)$ 在 $t \geq 0$ 上有定义。若存在 $\sigma > 0$ 使得积分
\begin{equation}
    F(s) = \mathcal{L}[f(t)] = \int_0^{+\infty} f(t)\mathrm{e}^{-st}\,\mathrm{d}t
\end{equation}
在 $\mathrm{Re}(s) > \sigma$ 内收敛，则称 $F(s)$ 为 $f(t)$ 的\textbf{拉普拉斯变换\index{拉普拉斯变换}}。

\begin{zhu}{}{}
我们约定 $f(t) = 0$（$t < 0$），这样拉普拉斯变换\index{拉普拉斯变换}可以看作傅里叶变换\index{傅里叶变换}的推广。
\end{zhu}
\end{dingliyi}

\subsection{拉普拉斯变换\index{拉普拉斯变换}的存在条件}

\begin{dingli}{}{存在定理}
若 $f(t)$ 满足：
\begin{enumerate}
    \item 在 $t \geq 0$ 的任意有限区间上分段连续
    \item 存在常数 $M > 0$ 和 $\sigma$，使得 $|f(t)| \leq M\mathrm{e}^{\sigma t}$（$t$ 充分大）
\end{enumerate}
则 $F(s) = \mathcal{L}[f(t)]$ 在 $\mathrm{Re}(s) > \sigma$ 内存在且解析。
\end{dingli}

\begin{proof}
由条件，$|f(t)| \leq M\mathrm{e}^{\sigma t}$，当 $\mathrm{Re}(s) > \sigma$ 时，
\begin{equation}
    \left|\int_0^{+\infty} f(t)\mathrm{e}^{-st}\,\mathrm{d}t\right| \leq M\int_0^{+\infty} \mathrm{e}^{-(\mathrm{Re}(s)-\sigma)t}\,\mathrm{d}t = \frac{M}{\mathrm{Re}(s)-\sigma} < +\infty
\end{equation}
故积分收敛。解析性可由积分号下求导验证（求导后积分仍收敛）。
\end{proof}
\begin{dingliyi}{}{增长指数}
满足 $|f(t)| \leq M\mathrm{e}^{\sigma t}$ 的最小 $\sigma$ 称为 $f(t)$ 的\textbf{增长指数}。
\end{dingliyi}

\subsection{常用函数的拉普拉斯变换\index{拉普拉斯变换}}

\begin{lizi}{}{}
求 $f(t) = 1$ 的拉普拉斯变换\index{拉普拉斯变换}。

\begin{jie}
    \begin{equation}
        \mathcal{L}[1] = \int_0^{+\infty} \mathrm{e}^{-st}\,\mathrm{d}t = \frac{1}{s}, \quad \mathrm{Re}(s) > 0
    \end{equation}
\end{jie}
\end{lizi}

\begin{lizi}{}{}
求 $f(t) = \mathrm{e}^{at}$（$a$ 为复数\index{复数}）的拉普拉斯变换\index{拉普拉斯变换}。

\begin{jie}
    \begin{equation}
        \mathcal{L}[\mathrm{e}^{at}] = \int_0^{+\infty} \mathrm{e}^{at}\mathrm{e}^{-st}\,\mathrm{d}t = \int_0^{+\infty} \mathrm{e}^{-(s-a)t}\,\mathrm{d}t = \frac{1}{s - a}, \quad \mathrm{Re}(s) > \mathrm{Re}(a)
    \end{equation}
\end{jie}
\end{lizi}

\begin{lizi}{}{}
求 $f(t) = \sin\omega t$ 和 $f(t) = \cos\omega t$ 的拉普拉斯变换\index{拉普拉斯变换}。

\begin{jie}
    利用欧拉公式：
    \begin{equation}
        \sin\omega t = \frac{\mathrm{e}^{\mathrm{i}\omega t} - \mathrm{e}^{-\mathrm{i}\omega t}}{2\mathrm{i}}, \quad \cos\omega t = \frac{\mathrm{e}^{\mathrm{i}\omega t} + \mathrm{e}^{-\mathrm{i}\omega t}}{2}
    \end{equation}
    
    所以
    \begin{align}
        \mathcal{L}[\sin\omega t] &= \frac{1}{2\mathrm{i}}\left(\frac{1}{s - \mathrm{i}\omega} - \frac{1}{s + \mathrm{i}\omega}\right) = \frac{\omega}{s^2 + \omega^2} \\
        \mathcal{L}[\cos\omega t] &= \frac{1}{2}\left(\frac{1}{s - \mathrm{i}\omega} + \frac{1}{s + \mathrm{i}\omega}\right) = \frac{s}{s^2 + \omega^2}
    \end{align}
\end{jie}
\end{lizi}

\begin{lizi}{}{}
求 $f(t) = t^n$（$n$ 为正整数）的拉普拉斯变换\index{拉普拉斯变换}。

\begin{jie}
    利用分部积分：
    \begin{equation}
        \mathcal{L}[t^n] = \int_0^{+\infty} t^n\mathrm{e}^{-st}\,\mathrm{d}t
    \end{equation}
    
    设 $u = t^n$，$\mathrm{d}v = \mathrm{e}^{-st}\,\mathrm{d}t$：
    \begin{equation}
        \mathcal{L}[t^n] = -\frac{t^n\mathrm{e}^{-st}}{s}\Big|_0^{+\infty} + \frac{n}{s}\int_0^{+\infty} t^{n-1}\mathrm{e}^{-st}\,\mathrm{d}t = \frac{n}{s}\mathcal{L}[t^{n-1}]
    \end{equation}
    
    递推得：
    \begin{equation}
        \mathcal{L}[t^n] = \frac{n!}{s^{n+1}}
    \end{equation}
\end{jie}
\end{lizi}

\section{拉普拉斯变换\index{拉普拉斯变换}的性质}

\subsection{基本性质}

\begin{dingli}{}{线性性质}
\begin{equation}
    \mathcal{L}[\alpha f(t) + \beta g(t)] = \alpha F(s) + \beta G(s)
\end{equation}
\end{dingli}

\begin{proof}
由积分的线性性，
\begin{equation}
    \mathcal{L}[\alpha f(t) + \beta g(t)] = \int_0^{+\infty} [\alpha f(t) + \beta g(t)]\mathrm{e}^{-st}\,\mathrm{d}t = \alpha F(s) + \beta G(s)
\end{equation}
\end{proof}
\begin{dingli}{}{延迟性质（时域平移）}
\begin{equation}
    \mathcal{L}[f(t-\tau)u(t-\tau)] = \mathrm{e}^{-\tau s}F(s)
\end{equation}
其中 $u(t)$ 是单位阶跃函数。
\end{dingli}

\begin{proof}
\begin{equation}
    \mathcal{L}[f(t-\tau)u(t-\tau)] = \int_{\tau}^{+\infty} f(t-\tau)\mathrm{e}^{-st}\,\mathrm{d}t = \mathrm{e}^{-\tau s}\int_0^{+\infty} f(u)\mathrm{e}^{-su}\,\mathrm{d}u = \mathrm{e}^{-\tau s}F(s)
\end{equation}
\end{proof}

\begin{dingli}{}{频移性质}
\begin{equation}
    \mathcal{L}[\mathrm{e}^{at}f(t)] = F(s-a)
\end{equation}
\end{dingli}

\begin{proof}
令 $u = at$，则
\begin{equation}
    \mathcal{L}[f(at)] = \int_0^{+\infty} f(at)\mathrm{e}^{-st}\,\mathrm{d}t = \frac{1}{a}\int_0^{+\infty} f(u)\mathrm{e}^{-\frac{s}{a}u}\,\mathrm{d}u = \frac{1}{a}F\left(\frac{s}{a}\right)
\end{equation}
\end{proof}
\begin{dingli}{}{伸缩性质}
\begin{equation}
    \mathcal{L}[f(at)] = \frac{1}{a}F\left(\frac{s}{a}\right), \quad a > 0
\end{equation}
\end{dingli}

\subsection{微分性质}

\begin{zhongyao}[时域微分性质]
设 $f(t)$ 连续，$f'(t)$ 分段连续，则
\[
    \mathcal{L}[f'(t)] = sF(s) - f(0)
\]
\end{zhongyao}

\begin{proof}
\begin{equation}
    \mathcal{L}[f'(t)] = \int_0^{+\infty} f'(t)\mathrm{e}^{-st}\,\mathrm{d}t = f(t)\mathrm{e}^{-st}\Big|_0^{+\infty} + s\int_0^{+\infty} f(t)\mathrm{e}^{-st}\,\mathrm{d}t
\end{equation}

由于 $|f(t)| \leq M\mathrm{e}^{\sigma t}$，当 $\mathrm{Re}(s) > \sigma$ 时，$f(t)\mathrm{e}^{-st} \to 0$（$t \to +\infty$）。

所以 $\mathcal{L}[f'(t)] = -f(0) + sF(s)$。
\end{proof}

\begin{dingli}{}{高阶导数}
\begin{equation}
    \mathcal{L}[f^{(n)}(t)] = s^n F(s) - s^{n-1}f(0) - s^{n-2}f'(0) - \cdots - f^{(n-1)}(0)
\end{equation}
\end{dingli}

\begin{proof}
对一阶导数公式反复应用。设 $\mathcal{L}[f'(t)] = sF(s) - f(0)$，则
\begin{equation}
    \mathcal{L}[f''(t)] = s\mathcal{L}[f'(t)] - f'(0) = s[sF(s) - f(0)] - f'(0) = s^2F(s) - sf(0) - f'(0)
\end{equation}
由数学归纳法可得一般公式。
\end{proof}
\begin{dingli}{}{频域微分}
\begin{equation}
    \mathcal{L}[tf(t)] = -F'(s)
\end{equation}
\begin{equation}
    \mathcal{L}[t^n f(t)] = (-1)^n F^{(n)}(s)
\end{equation}
\end{dingli}

\begin{proof}
由定义 $F(s) = \int_0^{+\infty} f(t)\mathrm{e}^{-st}\,\mathrm{d}t$，对 $s$ 求导：
\begin{equation}
    F'(s) = \int_0^{+\infty} f(t) \cdot (-t)\mathrm{e}^{-st}\,\mathrm{d}t = -\int_0^{+\infty} tf(t)\mathrm{e}^{-st}\,\mathrm{d}t = -\mathcal{L}[tf(t)]
\end{equation}
反复求导得 $\mathcal{L}[t^n f(t)] = (-1)^n F^{(n)}(s)$。
\end{proof}
\begin{tishi}[微分性质的意义]
时域微分性质使得拉普拉斯变换\index{拉普拉斯变换}能够自动处理初始条件，这是它优于傅里叶变换\index{傅里叶变换}的地方。
\end{tishi}

\subsection{积分性质}

\begin{dingli}{}{时域积分}
\begin{equation}
    \mathcal{L}\left[\int_0^t f(\tau)\,\mathrm{d}\tau\right] = \frac{F(s)}{s}
\end{equation}
\end{dingli}

\begin{proof}
设 $g(t) = \int_0^t f(\tau)\,\mathrm{d}\tau$，则 $g'(t) = f(t)$，$g(0) = 0$。由时域微分性质：
\begin{equation}
    \mathcal{L}[f(t)] = \mathcal{L}[g'(t)] = s\mathcal{L}[g(t)] - g(0) = s\mathcal{L}[g(t)]
\end{equation}
所以 $\mathcal{L}[g(t)] = \frac{F(s)}{s}$。
\end{proof}
\begin{dingli}{}{频域积分}
\begin{equation}
    \mathcal{L}\left[\frac{f(t)}{t}\right] = \int_s^{+\infty} F(u)\,\mathrm{d}u
\end{equation}
\end{dingli}

\begin{proof}
设 $g(t) = \frac{f(t)}{t}$，则 $f(t) = tg(t)$。由频域微分性质：
\begin{equation}
    F(s) = \mathcal{L}[tg(t)] = -G'(s)
\end{equation}
故 $G(s) = -\int F(s)\,\mathrm{d}s$。取积分限从 $s$ 到 $+\infty$，并注意到 $G(+\infty) = 0$，得
\begin{equation}
    G(s) = \int_s^{+\infty} F(u)\,\mathrm{d}u
\end{equation}
\end{proof}
\subsection{卷积定理}

\begin{dingliyi}{}{卷积}
\begin{equation}
    (f * g)(t) = \int_0^t f(\tau)g(t-\tau)\,\mathrm{d}\tau
\end{equation}
\end{dingliyi}

\begin{zhongyao}[卷积定理]
\[
    \mathcal{L}[f * g] = F(s)G(s)
\]
\end{zhongyao}

\section{拉普拉斯逆变换}

\subsection{反演公式}

\begin{dingli}{}{拉普拉斯逆变换}
\begin{equation}
    f(t) = \mathcal{L}^{-1}[F(s)] = \frac{1}{2\pi\mathrm{i}}\int_{\sigma-\mathrm{i}\infty}^{\sigma+\mathrm{i}\infty} F(s)\mathrm{e}^{st}\,\mathrm{d}s
\end{equation}
其中 $\sigma$ 大于 $F(s)$ 所有奇点的实部。
\end{dingli}

\begin{proof}
由傅里叶逆变换公式推导。设 $F(s) = \mathcal{L}[f(t)]$，令 $s = \sigma + \mathrm{i}\omega$，则
\begin{equation}
    F(\sigma + \mathrm{i}\omega) = \int_0^{+\infty} f(t)\mathrm{e}^{-(\sigma+\mathrm{i}\omega)t}\,\mathrm{d}t = \int_0^{+\infty} [f(t)\mathrm{e}^{-\sigma t}]\mathrm{e}^{-\mathrm{i}\omega t}\,\mathrm{d}t
\end{equation}
这是 $g(t) = f(t)\mathrm{e}^{-\sigma t}u(t)$ 的傅里叶变换。由傅里叶逆变换：
\begin{equation}
    f(t)\mathrm{e}^{-\sigma t} = \frac{1}{2\pi}\int_{-\infty}^{+\infty} F(\sigma + \mathrm{i}\omega)\mathrm{e}^{\mathrm{i}\omega t}\,\mathrm{d}\omega
\end{equation}
两边乘以 $\mathrm{e}^{\sigma t}$，并令 $s = \sigma + \mathrm{i}\omega$，$\mathrm{d}s = \mathrm{i}\,\mathrm{d}\omega$，得逆变换公式。
\end{proof}

\subsection{留数法求逆变换}

利用复变函数的留数理论，逆变换可以表示为：

\begin{dingli}{}{海维赛德展开定理}
设 $F(s) = \frac{A(s)}{B(s)}$ 是有理函数，$s_1, s_2, \ldots, s_n$ 是 $B(s)$ 的零点。

\begin{enumerate}
    \item 若 $s_k$ 是 $B(s)$ 的一阶零点，则
    \begin{equation}
        \operatorname{Res}[F(s)\mathrm{e}^{st}, s_k] = \frac{A(s_k)}{B'(s_k)}\mathrm{e}^{s_k t}
    \end{equation}
    
    \item 若 $s_k$ 是 $B(s)$ 的 $m$ 阶零点，则
    \begin{equation}
        \operatorname{Res}[F(s)\mathrm{e}^{st}, s_k] = \frac{1}{(m-1)!}\lim_{s \to s_k} \frac{\mathrm{d}^{m-1}}{\mathrm{d}s^{m-1}}\left[(s-s_k)^m F(s)\mathrm{e}^{st}\right]
    \end{equation}
\end{enumerate}

逆变换为：
\begin{equation}
    f(t) = \sum_{k=1}^{n} \operatorname{Res}[F(s)\mathrm{e}^{st}, s_k]
\end{equation}
\end{dingli}

\begin{proof}
由留数定理，拉普拉斯逆变换的积分路径可以补成闭曲线（左侧添加半圆弧），当半圆弧半径充分大时，其上的积分趋于零。因此
\begin{equation}
    f(t) = \frac{1}{2\pi\mathrm{i}}\int_{\sigma-\mathrm{i}\infty}^{\sigma+\mathrm{i}\infty} F(s)\mathrm{e}^{st}\,\mathrm{d}s = \sum_{k=1}^{n} \operatorname{Res}[F(s)\mathrm{e}^{st}, s_k]
\end{equation}
其中 $s_k$ 是 $F(s)\mathrm{e}^{st}$ 的所有极点。对于有理函数 $F(s) = A(s)/B(s)$，极点就是 $B(s)$ 的零点。

一阶极点的留数公式：
\begin{equation}
    \operatorname{Res}[F(s)\mathrm{e}^{st}, s_k] = \lim_{s \to s_k} (s-s_k)F(s)\mathrm{e}^{st} = \lim_{s \to s_k} \frac{A(s)}{B(s)/(s-s_k)}\mathrm{e}^{st} = \frac{A(s_k)}{B'(s_k)}\mathrm{e}^{s_k t}
\end{equation}
其中用到洛必达法则或 $B(s) \approx B'(s_k)(s-s_k)$。

$m$ 阶极点的留数公式直接由留数定义给出。
\end{proof}
\begin{lizi}{}{}
求 $F(s) = \frac{s}{s^2 + \omega^2}$ 的逆变换。

\begin{jie}
    极点：$s = \pm\mathrm{i}\omega$，都是一阶极点。
    
    \begin{align}
        \operatorname{Res}[F(s)\mathrm{e}^{st}, \mathrm{i}\omega] &= \frac{s}{2s}\mathrm{e}^{st}\Big|_{s=\mathrm{i}\omega} = \frac{1}{2}\mathrm{e}^{\mathrm{i}\omega t} \\
        \operatorname{Res}[F(s)\mathrm{e}^{st}, -\mathrm{i}\omega] &= \frac{s}{2s}\mathrm{e}^{st}\Big|_{s=-\mathrm{i}\omega} = \frac{1}{2}\mathrm{e}^{-\mathrm{i}\omega t}
    \end{align}
    
    所以
    \begin{equation}
        f(t) = \frac{1}{2}(\mathrm{e}^{\mathrm{i}\omega t} + \mathrm{e}^{-\mathrm{i}\omega t}) = \cos\omega t
    \end{equation}
\end{jie}
\end{lizi}

\subsection{部分分式法}

对于有理函数 $F(s) = \frac{A(s)}{B(s)}$，可以先分解为部分分式，再利用变换表求逆。

\begin{lizi}{}{}
求 $F(s) = \frac{s+3}{s(s+1)(s+2)}$ 的逆变换。

\begin{jie}
    设 $F(s) = \frac{A}{s} + \frac{B}{s+1} + \frac{C}{s+2}$。
    
    确定系数：
    \begin{align}
        A &= s \cdot F(s)\Big|_{s=0} = \frac{0+3}{(0+1)(0+2)} = \frac{3}{2} \\
        B &= (s+1) \cdot F(s)\Big|_{s=-1} = \frac{-1+3}{(-1)(-1+2)} = -2 \\
        C &= (s+2) \cdot F(s)\Big|_{s=-2} = \frac{-2+3}{(-2)(-2+1)} = \frac{1}{2}
    \end{align}
    
    所以
    \begin{equation}
        F(s) = \frac{3/2}{s} - \frac{2}{s+1} + \frac{1/2}{s+2}
    \end{equation}
    
    逆变换：
    \begin{equation}
        f(t) = \frac{3}{2} - 2\mathrm{e}^{-t} + \frac{1}{2}\mathrm{e}^{-2t}
    \end{equation}
\end{jie}
\end{lizi}

\section{拉普拉斯变换\index{拉普拉斯变换}的应用}

\subsection{求解常微分方程}

\begin{lizi}{}{}
求解初值问题：
\begin{equation}
    \begin{cases}
        y'' + 4y = 0 \\
        y(0) = 1, \; y'(0) = 0
    \end{cases}
\end{equation}

\begin{jie}
    设 $Y(s) = \mathcal{L}[y(t)]$。对方程两边作拉普拉斯变换\index{拉普拉斯变换}：
    \begin{equation}
        s^2 Y(s) - sy(0) - y'(0) + 4Y(s) = 0
    \end{equation}
    
    代入初始条件：
    \begin{equation}
        s^2 Y(s) - s + 4Y(s) = 0 \implies Y(s) = \frac{s}{s^2 + 4}
    \end{equation}
    
    逆变换：
    \begin{equation}
        y(t) = \cos 2t
    \end{equation}
\end{jie}
\end{lizi}

\begin{lizi}{}{}
求解初值问题：
\begin{equation}
    \begin{cases}
        y'' + 3y' + 2y = \mathrm{e}^{-t} \\
        y(0) = 0, \; y'(0) = 0
    \end{cases}
\end{equation}

\begin{jie}
    设 $Y(s) = \mathcal{L}[y(t)]$。变换后：
    \begin{equation}
        s^2 Y(s) + 3sY(s) + 2Y(s) = \frac{1}{s+1}
    \end{equation}
    
    解得：
    \begin{equation}
        Y(s) = \frac{1}{(s+1)^2(s+2)}
    \end{equation}
    
    部分分式分解：
    \begin{equation}
        Y(s) = \frac{1}{(s+1)^2} - \frac{1}{s+1} + \frac{1}{s+2}
    \end{equation}
    
    逆变换：
    \begin{equation}
        y(t) = t\mathrm{e}^{-t} - \mathrm{e}^{-t} + \mathrm{e}^{-2t} = (t-1)\mathrm{e}^{-t} + \mathrm{e}^{-2t}
    \end{equation}
\end{jie}
\end{lizi}

\subsection{求解偏微分方程}

\begin{lizi}{}{}
求解半无界热传导问题：
\begin{equation}
    \begin{cases}
        u_t = a^2 u_{xx}, & x > 0, \; t > 0 \\
        u(0, t) = u_0, & t > 0 \\
        u(x, 0) = 0, & x > 0
    \end{cases}
\end{equation}

\begin{jie}
    对 $t$ 进行拉普拉斯变换\index{拉普拉斯变换}，设 $U(x, s) = \mathcal{L}[u(x, t)]$。
    
    方程变为：
    \begin{equation}
        sU = a^2 U_{xx} \implies U_{xx} - \frac{s}{a^2}U = 0
    \end{equation}
    
    通解：
    \begin{equation}
        U(x, s) = A\mathrm{e}^{\sqrt{s}x/a} + B\mathrm{e}^{-\sqrt{s}x/a}
    \end{equation}
    
    边界条件：$U(0, s) = \mathcal{L}[u_0] = \frac{u_0}{s}$。

    当 $x \to +\infty$ 时，$U$ 有界，所以 $A = 0$。

    由 $U(0, s) = B = \frac{u_0}{s}$，得
    \begin{equation}
        U(x, s) = \frac{u_0}{s}\mathrm{e}^{-\sqrt{s}x/a}
    \end{equation}
    
    逆变换（利用拉普拉斯变换\index{拉普拉斯变换}表或留数计算）：
    \begin{equation}
        u(x, t) = u_0 \operatorname{erfc}\left(\frac{x}{2a\sqrt{t}}\right)
    \end{equation}
    其中 $\operatorname{erfc}$ 是补余误差函数。
\end{jie}
\end{lizi}

\subsection{电路分析}

\begin{wulibj}[应用：RLC 电路]
拉普拉斯变换\index{拉普拉斯变换}在电路分析中有重要应用。将电路元件变换到 $s$ 域：
\begin{itemize}
    \item 电阻：$V(s) = RI(s)$
    \item 电感：$V(s) = sLI(s) - Li(0)$
    \item 电容：$V(s) = \frac{1}{sC}I(s) + \frac{v(0)}{s}$
\end{itemize}

这样，电路的微分方程问题转化为代数方程问题。
\end{wulibj}

\textbf{基础题}

\begin{enumerate}
    \item 计算下列函数的拉普拉斯变换\index{拉普拉斯变换}：
    \begin{enumerate}
        \item $f(t) = t^2\mathrm{e}^{-t}$
        \item $f(t) = \sin 2t\cos t$
        \item $f(t) = u(t-1)\mathrm{e}^{-(t-1)}$
    \end{enumerate}
    
    \item 利用拉普拉斯变换\index{拉普拉斯变换}的性质，由 $\mathcal{L}[\sin t] = \frac{1}{s^2+1}$ 推导 $\mathcal{L}[t\sin t]$。
\end{enumerate}

\textbf{综合题}

\begin{enumerate}[resume]
    \item 求下列函数的拉普拉斯逆变换：
    \begin{enumerate}
        \item $F(s) = \frac{2s+1}{s(s+1)(s+2)}$
        \item $F(s) = \frac{1}{s^2(s^2+4)}$
    \end{enumerate}
    
    \item 用拉普拉斯变换\index{拉普拉斯变换}求解：
    \begin{equation}
        \begin{cases}
            y'' + 2y' + y = \sin t \\
            y(0) = 0, \; y'(0) = 1
        \end{cases}
    \end{equation}
\end{enumerate}

\textbf{挑战题}

\begin{enumerate}[resume]

    \item 设 $f(t)$ 满足积分方程
    \begin{equation}
        f(t) = t + \int_0^t f(\tau)\sin(t-\tau)\,\mathrm{d}\tau
    \end{equation}
    求 $f(t)$。
\end{enumerate}
