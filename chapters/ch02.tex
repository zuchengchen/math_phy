% 第二章：复变函数与解析函数初步

\chapter{复变函数与解析函数初步}

\begin{wulibj}[本章导读]
第一章里，我们已经把复数看成了复平面上的点、向量和旋转伸缩的载体。接下来，一个非常自然的问题就是：如果自变量和函数值都放到复平面上，我们得到的``函数''会是什么样子？

乍一看，这好像只是把高等数学里的 $x$ 改成了 $z$。但真正学下去就会发现，事情远没有这么简单。因为复数本身是二维的，所以复变函数的极限、连续和导数，都要同时面对``从任意方向趋近''这一要求。正是这一点，使复分析中的解析函数表现出比实变函数强得多的刚性。

本章要完成下面几件事：
\begin{enumerate}
    \item 建立复平面上点集、区域与曲线的语言，为后面的复积分与奇点讨论打基础；
    \item 理解复变函数的概念、极限、连续、导数与解析性，弄清楚复可导为何比实可导严格得多；
    \item 推导柯西--黎曼条件，并由此看见解析函数和调和函数、二维势场、局部保角性之间的联系。
\end{enumerate}

这三部分内容彼此是连着的。第一节建立的``区域''和``曲线''语言，会在下一章的复积分中立刻派上用场；本章后半部分得到的柯西--黎曼条件和调和性质，又会把复分析和后面的拉普拉斯方程、二维静电场自然接起来；而第一章里反复强调的``复数乘法对应旋转与伸缩''，会在本章最后以``导数的局部几何意义''的形式重新出现。
\end{wulibj}

\section{复平面上的点集、区域与曲线}

\begin{wulibj}[为什么先讲这一节]
在高等数学里，函数的定义域往往是实轴上的一个区间；而到了复变函数里，定义域变成了复平面中的点集。于是很多在一维里不太显眼的区别，现在都必须认真对待：什么叫``在点附近''？什么叫``边界上''？什么叫``区域里没有洞''？什么样的曲线才适合作为积分路径？

这些概念看起来像是术语准备，但其实它们都是后面理论的骨架。比如，柯西积分定理要用到单连通区域；奇点要用到去心邻域；复积分则要沿曲线进行。要是这一节的语言没有建立起来，后面很多定理都会显得像是突然从天上掉下来的。
\end{wulibj}

\subsection{邻域与去心邻域}

\begin{dingliyi}{邻域}{}
设 $z_0$ 是复平面上的一个点，$\varepsilon>0$。点集
\begin{equation}
    U(z_0,\varepsilon)=\{z\in\mathbb{C}:|z-z_0|<\varepsilon\}
\end{equation}
称为点 $z_0$ 的\textbf{$\varepsilon$ 邻域}，也就是以 $z_0$ 为圆心、$\varepsilon$ 为半径的开圆盘。
\end{dingliyi}

\begin{dingliyi}{去心邻域}{}
集合
\begin{equation}
    \{z\in\mathbb{C}:0<|z-z_0|<\varepsilon\}
\end{equation}
称为点 $z_0$ 的一个\textbf{去心邻域}。
\end{dingliyi}

\begin{zhu}{}{}
``去心''这两个字的意思很直白：就是把中心点 $z_0$ 本身拿掉。以后讨论极限时，我们关心的是 $z$ 靠近 $z_0$ 的过程，而不是 $z$ 恰好等于 $z_0$，所以去心邻域会频繁出现。
\end{zhu}

\subsection{内点、外点与边界点}

\begin{dingliyi}{内点、外点与边界点}{}
设 $E\subset\mathbb{C}$，$z_0\in\mathbb{C}$。
\begin{itemize}
    \item 若存在 $\varepsilon>0$，使得 $U(z_0,\varepsilon)\subset E$，则称 $z_0$ 是 $E$ 的\textbf{内点}；
    \item 若存在 $\varepsilon>0$，使得 $U(z_0,\varepsilon)\cap E=\emptyset$，则称 $z_0$ 是 $E$ 的\textbf{外点}；
    \item 若 $z_0$ 的任意邻域内既有 $E$ 中的点，也有不属于 $E$ 的点，则称 $z_0$ 是 $E$ 的\textbf{边界点}。
\end{itemize}
\end{dingliyi}

\begin{lizi}{判断点的类型}{}
设
\[
    E=\{z:|z|<1\}
\]
是单位开圆盘，判断下列各点的类型：
\[
    z_1=0,\qquad z_2=1,\qquad z_3=2.
\]

\begin{jie}
点 $z_1=0$ 是内点，因为只要取半径足够小的邻域，例如 $U(0,1/2)$，它就完全包含在 $E$ 内。

点 $z_2=1$ 是边界点。因为无论邻域取多小，它都会同时碰到圆盘内部的点和圆盘外部的点。

点 $z_3=2$ 是外点，因为可以找到一个小邻域，例如 $U(2,1/2)$，使它与 $E$ 完全不相交。
\end{jie}
\end{lizi}

\subsection{开集、闭集与区域}

\begin{dingliyi}{开集与闭集}{}
\begin{itemize}
    \item 若点集 $E$ 中每一点都是它的内点，则称 $E$ 是\textbf{开集}；
    \item 若 $\mathbb{C}\setminus E$ 是开集，则称 $E$ 是\textbf{闭集}；
    \item $E$ 的所有边界点构成的集合称为 $E$ 的\textbf{边界}，记作 $\partial E$。
\end{itemize}
\end{dingliyi}

\begin{zhu}{}{}
``开''和``闭''并不是互相对立的两种状态。空集 $\emptyset$ 和整个复平面 $\mathbb{C}$ 既是开集，也是闭集。
\end{zhu}

\begin{dingliyi}{区域}{}
若点集 $D\subset\mathbb{C}$ 同时满足：
\begin{enumerate}
    \item $D$ 是开集；
    \item $D$ 是连通的；
\end{enumerate}
则称 $D$ 为一个\textbf{区域}。
\end{dingliyi}

\begin{tishi}[为什么区域一定要要求``开'']
以后我们讨论函数在某点可导、在某区域内解析时，都会默认该点附近还有一小块空间可以活动。如果一个点集连边界点都拿来当内部点讨论，那么很多关于邻域、导数和积分路径的说法就会变得很别扭。所以复分析里通常把真正工作的对象选成开集，再额外单独处理边界。
\end{tishi}

\subsection{单连通与多连通}

连通只说明一片区域没有被分裂成互不相通的几块，但还没有说明它内部是不是有``洞''。这个区别在实分析里不太显眼，在复分析里却非常重要。

\begin{dingliyi}{单连通区域}{}
若区域 $D$ 内任意简单闭曲线都可以在 $D$ 内连续缩成一点，则称 $D$ 为\textbf{单连通区域}；否则称为\textbf{多连通区域}。
\end{dingliyi}

\begin{zhu}{}{}
直观地说，单连通区域就是``没有洞''的区域。像圆盘、上半平面、整个复平面都是单连通的；环域、去掉原点的平面则不是。
\end{zhu}

\begin{lizi}{判断区域是否单连通}{}
判断下列集合是否为单连通区域：
\[
    D_1=\{z:|z|<1\},\qquad D_2=\{z:1<|z|<2\},\qquad D_3=\mathbb{C}\setminus\{0\}.
\]

\begin{jie}
$D_1$ 是单位开圆盘，没有洞，所以是单连通区域。

$D_2$ 是环域，中间挖掉了一圈，所以不是单连通区域。

$D_3$ 虽然除了原点外其余部分都还连在一起，但原点的缺失形成了一个``绕不过去''的洞，因此它也是多连通的，不是单连通区域。
\end{jie}
\end{lizi}

\subsection{曲线、简单闭曲线与正向}

后面讲复积分时，我们不是在一块区域上直接求和，而是沿着某条路径积分。因此，必须先把``什么样的路径''说清楚。

这一组概念里，每个对象解决的问题其实不一样：``曲线''回答的是路径到底是什么；``分段光滑''保证后面参数化以后，积分表达式是好处理的；``简单闭曲线''和``正向''则是为闭路积分、柯西积分定理以及边界取向准备语言。把这几层意思先分清，后面就不会觉得这些术语是在集中堆砌。

\begin{dingliyi}{曲线}{}
若存在连续参数方程
\[
    z=z(t),\qquad a\le t\le b,
\]
则称点集 $C=\{z(t):a\le t\le b\}$ 为一条曲线。
\end{dingliyi}

\begin{dingliyi}{分段光滑曲线}{}
若参数方程在若干小区间上可微，且导数连续，则称该曲线为\textbf{分段光滑曲线}。
\end{dingliyi}

\begin{dingliyi}{简单闭曲线}{}
若曲线起点与终点重合，且除此之外没有自交，则称其为\textbf{简单闭曲线}。
\end{dingliyi}

\begin{dingliyi}{正向}{}
若简单闭曲线按逆时针方向绕行，使区域始终位于行进方向的左侧，则称该方向为\textbf{正向}。
\end{dingliyi}

\begin{tishi}[为什么这里要强调正向]
以后写闭路积分
\[
    \oint_C f(z)\,\mathrm{d}z
\]
时，方向不是可有可无的小细节。把曲线反向以后，积分值会变号；而多连通区域里的外边界和内边界，取向也不一样。这些规则都会在下一章反复出现。
\end{tishi}

\section{复变函数的概念与映射观点}

\begin{wulibj}[为什么这一节不只是写个定义]
在实变函数里，我们习惯把函数画成一条曲线 $y=f(x)$；但到了复变函数里，这种图像已经不够用了。因为自变量 $z$ 本身就在复平面里，函数值 $w=f(z)$ 也落在另一个复平面里，所以更自然的理解方式，不是画一条曲线，而是把整个区域看成被映到另一个区域中去。

因此，这一节的任务有两个：一是建立复变函数最基本的语言，二是尽早让学生形成``映射''的眼光。后面讲局部保角性时，真正起作用的正是这个观点。
\end{wulibj}

\subsection{从实变函数到复变函数}

\begin{dingliyi}{复变函数}{}
设 $D\subset\mathbb{C}$。若对每个 $z\in D$，都按某一规则唯一对应一个复数 $w$，则称 $w$ 是 $z$ 的\textbf{复变函数}，记作
\begin{equation}
    w=f(z).
\end{equation}
\end{dingliyi}

\begin{zhu}{}{}
本书如无特别说明，默认讨论的是单值函数。像对数函数、根式函数这类多值对象，会在后面的选学章专门讨论。
\end{zhu}

\subsection{实部与虚部的分离}

设
\[
    z=x+\mathrm{i}y,\qquad w=f(z)=u+\mathrm{i}v.
\]
由于 $u$ 和 $v$ 最终都依赖于 $x,y$，所以复变函数通常可以写成
\begin{equation}
    f(z)=u(x,y)+\mathrm{i}v(x,y),
\end{equation}
其中 $u(x,y)$ 和 $v(x,y)$ 都是二元实函数，分别称为 $f(z)$ 的实部和虚部。

\begin{lizi}{把复函数写成 $u+\mathrm{i}v$ 的形式}{}
设 $f(z)=z^2$，求它的实部和虚部。

\begin{jie}
设 $z=x+\mathrm{i}y$，则
\begin{align*}
    f(z)&=(x+\mathrm{i}y)^2\\
        &=x^2+2\mathrm{i}xy+\mathrm{i}^2y^2\\
        &=(x^2-y^2)+\mathrm{i}(2xy).
\end{align*}
因此
\[
    u(x,y)=x^2-y^2,\qquad v(x,y)=2xy.
\]
\end{jie}
\end{lizi}

\begin{jinshi}[一个最容易出错的小地方]
这里的虚部是实函数 $v(x,y)$，不是 $\mathrm{i}v(x,y)$。例如，若
\[
    f(z)=3-2\mathrm{i},
\]
那么虚部是 $-2$，不是 $-2\mathrm{i}$。
\end{jinshi}

\subsection{几个最基本的例子}

\begin{lizi}{一个带定义域限制的复函数}{}
把
\[
    f(z)=\frac{1}{z}
\]
写成 $u(x,y)+\mathrm{i}v(x,y)$ 的形式，并指出它的定义域。

\begin{jie}
设 $z=x+\mathrm{i}y$，则
\[
    \frac{1}{z}=\frac{1}{x+\mathrm{i}y}\cdot\frac{x-\mathrm{i}y}{x-\mathrm{i}y}
    =\frac{x-\mathrm{i}y}{x^2+y^2}.
\]
因此
\[
    u(x,y)=\frac{x}{x^2+y^2},\qquad v(x,y)=-\frac{y}{x^2+y^2}.
\]

但这里要求
\[
    x^2+y^2\neq 0,
\]
也就是 $z\neq 0$。所以它的定义域是 $\mathbb{C}\setminus\{0\}$。
\end{jie}
\end{lizi}

\begin{lizi}{一个看起来很简单但后面会出问题的例子}{}
设
\[
    f(z)=\bar z.
\]
写出它的实部与虚部，并说明它的几何意义。

\begin{jie}
若 $z=x+\mathrm{i}y$，则
\[
    \bar z=x-\mathrm{i}y.
\]
因此
\[
    u(x,y)=x,\qquad v(x,y)=-y.
\]

几何上，这个映射把复平面上的点关于实轴做镜像反射。
\end{jie}
\end{lizi}

\section{复变函数的极限与连续}

\begin{wulibj}[为什么复变极限更严格]
实变函数里，$x\to x_0$ 的过程本质上只是在数轴上从左边或右边靠近；而复变函数里，$z\to z_0$ 却意味着可以从整个平面上的任意方向、沿任意路径靠近。这就是复变极限比实变极限严格得多的根本原因。

所以，这一节真正要讲清楚的，不是定义的外壳，而是定义的内核：\textbf{不管你怎样靠近同一个点，函数值都必须朝着同一个极限走去。} 一旦这句话在脑子里立住了，后面复导数为什么这么苛刻，就会变得顺理成章。
\end{wulibj}

\subsection{极限的定义}

\begin{dingliyi}{复变函数的极限}{}
设 $f(z)$ 在点 $z_0$ 的某个去心邻域内有定义。若存在复数 $A$，使得对于任意 $\varepsilon>0$，都存在 $\delta>0$，当
\begin{equation}
    0<|z-z_0|<\delta
\end{equation}
时，都有
\begin{equation}
    |f(z)-A|<\varepsilon,
\end{equation}
则称 $f(z)$ 当 $z\to z_0$ 时的极限为 $A$，记作
\begin{equation}
    \lim_{z\to z_0}f(z)=A.
\end{equation}
\end{dingliyi}

\subsection{为什么必须考虑所有路径}

定义中最值得注意的地方，是 $z\to z_0$ 并没有规定路径。也就是说，$z$ 可以沿直线、圆弧、折线，甚至更复杂的曲线趋近于 $z_0$。如果这些不同路径给出的结果不一致，那么极限就不存在。

\begin{lizi}{用两条路径否定极限存在}{}
讨论
\[
    f(z)=\frac{\bar z}{z}
\]
当 $z\to 0$ 时极限是否存在。

\begin{jie}
先沿实轴趋近。此时 $z=x$，且 $x\neq 0$，于是
\[
    \frac{\bar z}{z}=\frac{x}{x}=1.
\]
因此沿实轴趋近时，函数值趋于 $1$。

再沿虚轴趋近。此时 $z=\mathrm{i}y$，且 $y\neq 0$，于是
\[
    \frac{\bar z}{z}=\frac{-\mathrm{i}y}{\mathrm{i}y}=-1.
\]
因此沿虚轴趋近时，函数值趋于 $-1$。

两条路径得到的结果不同，所以
\[
    \lim_{z\to 0}\frac{\bar z}{z}
\]
不存在。
\end{jie}
\end{lizi}

\begin{tishi}[这一类题目的基本判断策略]
要证明极限不存在，最常用的方法不是从定义硬推，而是主动去找两条路径。只要找到两条路径使函数值趋于不同结果，结论就已经成立了。
\end{tishi}

\begin{lizi}{正面证明一个极限存在}{}
证明
\[
    \lim_{z\to 0} z\bar z=0.
\]

\begin{jie}
注意到
\[
    z\bar z=|z|^2.
\]
因此
\[
    |z\bar z-0|=|z|^2.
\]

现在用定义来证明极限存在。任给 $\varepsilon>0$，只要取
\[
    \delta=\sqrt{\varepsilon},
\]
那么当
\[
    0<|z|<\delta
\]
时，就有
\[
    |z\bar z-0|=|z|^2<\delta^2=\varepsilon.
\]
这正是
\[
    \lim_{z\to 0} z\bar z=0
\]
的定义。

这个例子说明：虽然复变极限必须对任意路径一致，但只要能找到一个和路径无关的估计，证明极限存在仍然可以非常直接。
\end{jie}
\end{lizi}

\subsection{极限的实部虚部表示}

前面我们强调的是复变极限的几何本质：必须对一切路径都一致。接下来换一个更适合计算的视角。既然复变函数可以写成
\[
    f(z)=u(x,y)+\mathrm{i}v(x,y),
\]
那么极限问题也应该能翻译成两个二元实函数的极限问题。这个翻译在后面讨论连续、可导和柯西--黎曼条件时都会非常有用。

\begin{dingli}{极限的实部虚部判据}{}\label{thm:ch02-limit-real-imag}
设
\[
    f(z)=u(x,y)+\mathrm{i}v(x,y),\qquad z_0=x_0+\mathrm{i}y_0,\qquad A=a+\mathrm{i}b.
\]
则
\[
    \lim_{z\to z_0}f(z)=A
\]
当且仅当
\[
    \lim_{(x,y)\to(x_0,y_0)}u(x,y)=a,\qquad
    \lim_{(x,y)\to(x_0,y_0)}v(x,y)=b.
\]
\end{dingli}

\begin{proof}
先证必要性。若
\[
    \lim_{z\to z_0}f(z)=A,
\]
则当 $z$ 足够接近 $z_0$ 时，
\[
    |f(z)-A|
    =\sqrt{(u-a)^2+(v-b)^2}
\]
可以任意小。于是
\[
    |u-a|\le |f(z)-A|,\qquad |v-b|\le |f(z)-A|.
\]
因此
\[
    u(x,y)\to a,\qquad v(x,y)\to b.
\]

再证充分性。若
\[
    u(x,y)\to a,\qquad v(x,y)\to b,
\]
则对任意 $\varepsilon>0$，存在 $\delta>0$，当 $0<|z-z_0|<\delta$ 时，
\[
    |u-a|<\frac{\varepsilon}{\sqrt{2}},\qquad |v-b|<\frac{\varepsilon}{\sqrt{2}}.
\]
于是
\[
    |f(z)-A|
    =\sqrt{(u-a)^2+(v-b)^2}
    <\sqrt{\frac{\varepsilon^2}{2}+\frac{\varepsilon^2}{2}}
    =\varepsilon.
\]
所以
\[
    \lim_{z\to z_0}f(z)=A.
\]
\end{proof}

\subsection{连续性}

\begin{dingliyi}{连续}{}
若函数 $f(z)$ 在点 $z_0$ 的某个邻域内有定义，且
\begin{equation}
    \lim_{z\to z_0}f(z)=f(z_0),
\end{equation}
则称 $f(z)$ 在点 $z_0$ \textbf{连续}。
\end{dingliyi}

\begin{dingli}{连续的实部虚部判据}{}
设
\[
    f(z)=u(x,y)+\mathrm{i}v(x,y).
\]
则 $f(z)$ 在 $z_0=x_0+\mathrm{i}y_0$ 处连续，当且仅当 $u(x,y)$ 与 $v(x,y)$ 在 $(x_0,y_0)$ 处连续。
\end{dingli}

\begin{proof}
这只是定理 \ref{thm:ch02-limit-real-imag} 的直接应用。因为 $f$ 在 $z_0$ 处连续，等价于
\[
    \lim_{z\to z_0}f(z)=f(z_0)=u(x_0,y_0)+\mathrm{i}v(x_0,y_0),
\]
再把极限拆成实部和虚部即可。
\end{proof}

\begin{dingli}{连续函数的基本性质}{}
若 $f(z)$ 与 $g(z)$ 在 $z_0$ 处连续，则
\begin{enumerate}
    \item $f(z)\pm g(z)$ 在 $z_0$ 处连续；
    \item $f(z)g(z)$ 在 $z_0$ 处连续；
    \item 若 $g(z_0)\neq 0$，则 $\dfrac{f(z)}{g(z)}$ 在 $z_0$ 处连续。
\end{enumerate}
\end{dingli}

\begin{lizi}{讨论有理函数的连续性}{}
讨论函数
\[
    f(z)=\frac{z}{z-\mathrm{i}}
\]
的连续性。

\begin{jie}
分子 $z$ 和分母 $z-\mathrm{i}$ 都是复平面上的连续函数，因此在分母不为零的地方，商函数也连续。这里分母在
\[
    z=\mathrm{i}
\]
处为零，所以 $f(z)$ 在 $\mathbb{C}\setminus\{\mathrm{i}\}$ 内连续，而在 $z=\mathrm{i}$ 处无定义，自然也谈不上连续。
\end{jie}
\end{lizi}

\section{复变函数的导数与解析性}

\begin{wulibj}[为什么复导数会比实导数苛刻得多]
实变函数的导数讨论的是
\[
    \frac{f(x+\Delta x)-f(x)}{\Delta x}
\]
在 $\Delta x\to 0$ 时的极限。由于 $\Delta x$ 只能沿实轴趋近于零，所以我们最终只是在一维里比较左右两边的变化率。

而复变函数里，增量
\[
    \Delta z=\Delta x+\mathrm{i}\Delta y
\]
可以从任意方向趋近于零。于是，差商极限不仅要存在，还必须与趋近方向无关。这一下，要求就比实变函数严格得多了。本节要做的，就是把这一点真正讲透。
\end{wulibj}

\subsection{复导数的定义}

\begin{dingliyi}{复导数}{}
设函数 $f(z)$ 在点 $z_0$ 的某个邻域内有定义。若极限
\begin{equation}
    \lim_{\Delta z\to 0}\frac{f(z_0+\Delta z)-f(z_0)}{\Delta z}
\end{equation}
存在，则称 $f(z)$ 在 $z_0$ \textbf{可导}，并称该极限为 $f$ 在 $z_0$ 处的\textbf{导数}，记作 $f'(z_0)$。
\end{dingliyi}

\begin{jinshi}[这里真正严格的地方在哪里]
在复变函数里，$\Delta z\to 0$ 不是只沿某一条轴靠近，而是可以从整个复平面上的任意方向趋近于零。于是，差商极限必须对所有趋近方向都给出同一个结果。这就是复可导比实可导严格得多的根源。
\end{jinshi}

\subsection{由定义直接求导}

\begin{lizi}{用定义求 $z^2$ 的导数}{}
设 $f(z)=z^2$，求 $f'(z)$。

\begin{jie}
由定义，
\begin{align*}
    f'(z)
    &=\lim_{\Delta z\to 0}\frac{(z+\Delta z)^2-z^2}{\Delta z}\\
    &=\lim_{\Delta z\to 0}\frac{2z\Delta z+(\Delta z)^2}{\Delta z}\\
    &=\lim_{\Delta z\to 0}(2z+\Delta z)\\
    &=2z.
\end{align*}
所以
\[
    f'(z)=2z.
\]
\end{jie}
\end{lizi}

\subsection{一个处处不可导的经典反例}

\begin{lizi}{证明 $f(z)=\bar z$ 处处不可导}{}
设 $f(z)=\bar z$，证明它在任意点都不可导。

\begin{jie}
取任意点 $z$，则
\[
    \frac{f(z+\Delta z)-f(z)}{\Delta z}
    =\frac{\overline{z+\Delta z}-\bar z}{\Delta z}
    =\frac{\overline{\Delta z}}{\Delta z}.
\]

若 $\Delta z$ 沿实轴趋近于零，即 $\Delta z=\Delta x$，则
\[
    \frac{\overline{\Delta z}}{\Delta z}
    =\frac{\Delta x}{\Delta x}=1.
\]

若 $\Delta z$ 沿虚轴趋近于零，即 $\Delta z=\mathrm{i}\Delta y$，则
\[
    \frac{\overline{\Delta z}}{\Delta z}
    =\frac{-\mathrm{i}\Delta y}{\mathrm{i}\Delta y}=-1.
\]

两条路径给出的结果不同，因此导数极限不存在。由于 $z$ 是任意点，所以 $f(z)=\bar z$ 在任意点都不可导。
\end{jie}
\end{lizi}

\begin{tishi}[这个反例为什么重要]
$\bar z$ 看上去非常光滑，几何上也只是一个镜像反射，但它仍然处处不可导。这说明复可导要求的不是``看起来平滑''，而是局部变化率必须具有极强的一致性。
\end{tishi}

\subsection{解析、整函数与奇点}

有了``在一点可导''的概念以后，一个自然的问题就是：我们真正想研究的，到底只是某一个点的差商极限，还是函数在这一点附近整整一小块区域里的行为？复分析真正关心的，是后一种更强的性质，这就是``解析''要表达的意思。

也正因为如此，这里顺手把两个以后会不断出现的名字先立起来：若这种好性质扩展到整个复平面，就叫整函数；若在某点失效，就把那个点叫作奇点。

\begin{dingliyi}{解析函数}{}
若函数 $f(z)$ 在点 $z_0$ 的某个邻域内处处可导，则称 $f(z)$ 在 $z_0$ \textbf{解析}。若 $f(z)$ 在某区域 $D$ 内每一点都解析，则称 $f$ 为 $D$ 内的\textbf{解析函数}。
\end{dingliyi}

\begin{dingliyi}{整函数与奇点}{}
若函数在整个复平面上解析，则称其为\textbf{整函数}。若函数在某点不解析，则称该点为它的\textbf{奇点}。
\end{dingliyi}

\begin{zhu}{}{}
``在一点可导''和``在这一点解析''并不是同一回事。前者只讨论一个点上的导数极限，后者要求在这一点附近的一整小块区域里都可导。以后遇到这两个说法时，一定要分清。
\end{zhu}

\begin{lizi}{一个在一点可导但不解析的例子}{}
设
\[
    f(z)=|z|^2.
\]
讨论它在 $z=0$ 处是否可导，并说明它为什么不是解析函数。

\begin{jie}
先看 $z=0$ 处的导数。由定义，
\[
    \frac{f(0+\Delta z)-f(0)}{\Delta z}
    =\frac{|\Delta z|^2}{\Delta z}
    =\overline{\Delta z}.
\]
当 $\Delta z\to 0$ 时，
\[
    \overline{\Delta z}\to 0,
\]
所以 $f$ 在 $z=0$ 处可导，且
\[
    f'(0)=0.
\]

但若 $z\neq 0$，则
\[
    f(z)=z\bar z,
\]
它会同时依赖 $z$ 和 $\bar z$。事实上，后面用柯西--黎曼条件可以验证，它在 $z\neq 0$ 处不可导。因此 $f$ 不可能在 $0$ 的某个邻域内处处可导，所以它在 $z=0$ 处并不解析。
\end{jie}
\end{lizi}

\subsection{可导一定连续}

既然已经定义了可导，接下来一个很自然的问题就是：它和连续到底是什么关系？在高等数学里，大家已经知道``可导比连续更强''；复变函数里也是一样，而且这条结论马上就要在求导法则里派上用场。

\begin{dingli}{可导必连续}{}
若函数 $f(z)$ 在点 $z_0$ 可导，则它在 $z_0$ 处连续。
\end{dingli}

\begin{proof}
当 $z\neq z_0$ 时，可写
\[
    f(z)-f(z_0)
    =\frac{f(z)-f(z_0)}{z-z_0}(z-z_0).
\]
因为 $f$ 在 $z_0$ 可导，所以当 $z\to z_0$ 时，
\[
    \frac{f(z)-f(z_0)}{z-z_0}\to f'(z_0).
\]
特别地，这个差商在 $z_0$ 附近是有界的。另一方面，
\[
    z-z_0\to 0.
\]
因此右边是``有界量乘以趋于零的量''，于是
\[
    f(z)-f(z_0)\to 0.
\]
也就是说，
\[
    \lim_{z\to z_0}f(z)=f(z_0),
\]
所以 $f$ 在 $z_0$ 处连续。
\end{proof}

\begin{tishi}[为什么这条结论值得单独记住]
以后推导乘法法则、商法则，以及第三章讨论原函数和积分时，都会反复用到``可导先推出连续''这一点。要特别注意：方向只能是
\[
    \text{可导}\Longrightarrow \text{连续},
\]
反过来一般并不成立。
\end{tishi}

\subsection{基本求导法则}

\begin{dingli}{基本求导法则}{}
若 $f(z)$ 和 $g(z)$ 在点 $z$ 处可导，则
\begin{align}
    [f(z)\pm g(z)]'&=f'(z)\pm g'(z),\\
    [f(z)g(z)]'&=f'(z)g(z)+f(z)g'(z).
\end{align}
若此外 $g(z)\neq 0$，则
\begin{equation}
    \left[\frac{f(z)}{g(z)}\right]'
    =\frac{f'(z)g(z)-f(z)g'(z)}{[g(z)]^2}.
\end{equation}
\end{dingli}

\begin{proof}
加法法则直接由极限的线性性质得到。这里证明乘法法则：
\begin{align*}
    [f(z)g(z)]'
    &=\lim_{\Delta z\to 0}\frac{f(z+\Delta z)g(z+\Delta z)-f(z)g(z)}{\Delta z}\\
    &=\lim_{\Delta z\to 0}\frac{f(z+\Delta z)g(z+\Delta z)-f(z)g(z+\Delta z)}{\Delta z}\\
    &\quad+\lim_{\Delta z\to 0}\frac{f(z)g(z+\Delta z)-f(z)g(z)}{\Delta z}\\
    &=\lim_{\Delta z\to 0}g(z+\Delta z)\frac{f(z+\Delta z)-f(z)}{\Delta z}
      +f(z)\lim_{\Delta z\to 0}\frac{g(z+\Delta z)-g(z)}{\Delta z}.
\end{align*}
因为可导推出连续，所以 $g(z+\Delta z)\to g(z)$。于是
\[
    [f(z)g(z)]'=g(z)f'(z)+f(z)g'(z).
\]
商法则可把 $1/g(z)$ 先写出来再用乘法法则得到。
\end{proof}

\section{柯西--黎曼条件与可导判别}

\begin{wulibj}[为什么需要这一节]
仅靠导数定义来判断一个函数是否可导，往往非常麻烦。因为每次都要重新处理
\[
    \frac{f(z+\Delta z)-f(z)}{\Delta z}
\]
在任意方向上的极限。于是一个自然想法就是：能不能把``可导''翻译成实部 $u(x,y)$ 与虚部 $v(x,y)$ 的偏导关系？

柯西--黎曼条件正是这种想法的结果。它不是凭空出现的公式，而是复导数必须与趋近方向无关这一事实的直接表达。所以，这一节最重要的不是记住等式，而是看清这些等式是怎么来的。
\end{wulibj}

\subsection{柯西--黎曼条件的导出}

设
\[
    f(z)=u(x,y)+\mathrm{i}v(x,y),\qquad z=x+\mathrm{i}y.
\]
如果 $f$ 在点 $z$ 可导，那么导数
\[
    f'(z)=\lim_{\Delta z\to 0}\frac{f(z+\Delta z)-f(z)}{\Delta z}
\]
必须与 $\Delta z\to 0$ 的方式无关。

先让 $\Delta z=\Delta x$，也就是沿实轴方向趋近于零。此时
\begin{align*}
    f'(z)
    &=\lim_{\Delta x\to 0}
    \frac{u(x+\Delta x,y)-u(x,y)+\mathrm{i}[v(x+\Delta x,y)-v(x,y)]}{\Delta x}\\
    &=u_x+\mathrm{i}v_x.
\end{align*}

再让 $\Delta z=\mathrm{i}\Delta y$，也就是沿虚轴方向趋近于零。此时
\begin{align*}
    f'(z)
    &=\lim_{\Delta y\to 0}
    \frac{u(x,y+\Delta y)-u(x,y)+\mathrm{i}[v(x,y+\Delta y)-v(x,y)]}{\mathrm{i}\Delta y}\\
    &=\frac{1}{\mathrm{i}}\left(u_y+\mathrm{i}v_y\right)\\
    &=v_y-\mathrm{i}u_y.
\end{align*}

若 $f'(z)$ 存在，这两个结果必须相同，因此
\[
    u_x+\mathrm{i}v_x=v_y-\mathrm{i}u_y.
\]
比较实部与虚部，就得到
\begin{equation}\label{eq:ch02-cr}
    u_x=v_y,\qquad u_y=-v_x.
\end{equation}

\begin{dingli}{柯西--黎曼条件的必要性}{}
若
\[
    f(z)=u(x,y)+\mathrm{i}v(x,y)
\]
在点 $z$ 可导，则 $u$ 和 $v$ 在该点满足柯西--黎曼条件
\[
    u_x=v_y,\qquad u_y=-v_x.
\]
\end{dingli}

\subsection{偏导连续时的充分条件}

\begin{dingli}{可导的充分条件}{}\label{thm:ch02-cr-sufficient}
设
\[
    f(z)=u(x,y)+\mathrm{i}v(x,y)
\]
在点 $z_0=x_0+\mathrm{i}y_0$ 的某个邻域内有定义。若 $u_x,u_y,v_x,v_y$ 在 $(x_0,y_0)$ 附近存在并连续，且在 $(x_0,y_0)$ 处满足柯西--黎曼条件，则 $f$ 在 $z_0$ 可导。
\end{dingli}

\begin{proof}
设
\[
    \Delta z=\Delta x+\mathrm{i}\Delta y.
\]
由多元函数的全增量公式，
\begin{align*}
    \Delta u&=u_x\Delta x+u_y\Delta y+o(|\Delta z|),\\
    \Delta v&=v_x\Delta x+v_y\Delta y+o(|\Delta z|).
\end{align*}
于是
\begin{align*}
    \Delta f
    &=\Delta u+\mathrm{i}\Delta v\\
    &=(u_x\Delta x+u_y\Delta y)+\mathrm{i}(v_x\Delta x+v_y\Delta y)+o(|\Delta z|).
\end{align*}
由柯西--黎曼条件 $u_x=v_y,\ u_y=-v_x$，得
\begin{align*}
    \Delta f
    &=(u_x\Delta x-v_x\Delta y)+\mathrm{i}(v_x\Delta x+u_x\Delta y)+o(|\Delta z|)\\
    &=(u_x+\mathrm{i}v_x)(\Delta x+\mathrm{i}\Delta y)+o(|\Delta z|)\\
    &=(u_x+\mathrm{i}v_x)\Delta z+o(|\Delta z|).
\end{align*}
因此
\[
    \frac{\Delta f}{\Delta z}=u_x+\mathrm{i}v_x+\frac{o(|\Delta z|)}{\Delta z}.
\]
当 $\Delta z\to 0$ 时，最后一项趋于零，所以极限存在，并且
\[
    f'(z_0)=u_x(x_0,y_0)+\mathrm{i}v_x(x_0,y_0).
\]
故 $f$ 在 $z_0$ 可导。
\end{proof}

\begin{dingli}{导数公式}{}
在定理 \ref{thm:ch02-cr-sufficient} 的条件下，
\begin{equation}
    f'(z)=u_x+\mathrm{i}v_x=v_y-\mathrm{i}u_y.
\end{equation}
\end{dingli}

\subsection{典型例子}

\begin{lizi}{用 C-R 条件验证 $z^2$ 的可导性}{}
验证
\[
    f(z)=z^2=(x^2-y^2)+\mathrm{i}(2xy)
\]
在整个复平面上解析，并求其导数。

\begin{jie}
这里
\[
    u=x^2-y^2,\qquad v=2xy.
\]
于是
\[
    u_x=2x,\qquad u_y=-2y,\qquad v_x=2y,\qquad v_y=2x.
\]
因此
\[
    u_x=v_y,\qquad u_y=-v_x,
\]
柯西--黎曼条件处处成立，而且这些偏导数都连续，所以 $f(z)=z^2$ 在整个复平面上解析。

导数为
\[
    f'(z)=u_x+\mathrm{i}v_x=2x+\mathrm{i}(2y)=2z.
\]
\end{jie}
\end{lizi}

\begin{lizi}{再看一次 $\bar z$}{}
用柯西--黎曼条件验证 $f(z)=\bar z$ 在任意点都不可导。

\begin{jie}
由
\[
    \bar z=x-\mathrm{i}y
\]
可知
\[
    u=x,\qquad v=-y.
\]
于是
\[
    u_x=1,\qquad u_y=0,\qquad v_x=0,\qquad v_y=-1.
\]
显然
\[
    u_x\neq v_y.
\]
因此柯西--黎曼条件不成立，所以 $f(z)=\bar z$ 在任意点都不可导。这与上一节用导数定义得到的结论完全一致。
\end{jie}
\end{lizi}

\subsection{不要机械套用 C-R 条件}

\begin{jinshi}[满足 C-R 条件不一定就可导]
柯西--黎曼条件是可导的必要条件；若再配合偏导连续等条件，它也是充分条件。但如果只知道某一点上恰好满足 C-R 条件，这还不足以推出函数在该点可导。
\end{jinshi}

\begin{lizi}{一个在一点满足 C-R 条件但不可导的反例}{}
定义
\[
    f(z)=
    \begin{cases}
        \dfrac{x^3}{x^2+y^2}+\mathrm{i}\dfrac{y^3}{x^2+y^2}, & z\neq 0,\\[1em]
        0, & z=0.
    \end{cases}
\]
证明它在原点满足柯西--黎曼条件，但在原点不可导。

\begin{jie}
先算原点的偏导数。设
\[
    u(x,y)=
    \begin{cases}
        \dfrac{x^3}{x^2+y^2}, & (x,y)\neq(0,0),\\
        0, & (x,y)=(0,0),
    \end{cases}
\]
以及
\[
    v(x,y)=
    \begin{cases}
        \dfrac{y^3}{x^2+y^2}, & (x,y)\neq(0,0),\\
        0, & (x,y)=(0,0).
    \end{cases}
\]

由定义可得
\[
    u_x(0,0)=1,\qquad u_y(0,0)=0,\qquad v_x(0,0)=0,\qquad v_y(0,0)=1.
\]
所以在原点有
\[
    u_x=v_y,\qquad u_y=-v_x.
\]
也就是说，柯西--黎曼条件在原点形式上成立。

但看导数差商：
\[
    \frac{f(z)-f(0)}{z}=\frac{f(z)}{z}.
\]
若沿实轴 $y=0$ 趋近于零，则
\[
    f(x)=x,\qquad \frac{f(x)}{x}=1.
\]
若沿直线 $y=x$ 趋近于零，则
\[
    f(x+\mathrm{i}x)
    =\frac{x^3}{2x^2}+\mathrm{i}\frac{x^3}{2x^2}
    =\frac{x}{2}(1+\mathrm{i}),
\]
从而
\[
    \frac{f(x+\mathrm{i}x)}{x+\mathrm{i}x}=\frac{1}{2}.
\]
两条路径给出的结果不同，所以导数极限不存在。故 $f$ 在原点不可导。
\end{jie}
\end{lizi}

\section{解析函数的调和性质与共轭调和函数}

\begin{wulibj}[为什么这一节很重要]
如果只把柯西--黎曼条件看成一个``判断可导的技巧''，那它的作用就被低估了。事实上，C-R 条件一旦成立，就会强迫实部和虚部满足拉普拉斯方程。也就是说，解析函数内部并不是任意拼出来的，而是自带非常整齐的结构。

对数学物理方法来说，这一点尤其关键。因为拉普拉斯方程正是二维静电场、稳态温度场、理想流体等模型里最常见的方程之一。本节就是要让学生第一次看见：解析函数和物理中的势函数，并不是两套互不相干的语言。
\end{wulibj}

\subsection{调和函数}

\begin{dingliyi}{调和函数}{}
若二元实函数 $u(x,y)$ 满足
\begin{equation}
    u_{xx}+u_{yy}=0,
\end{equation}
则称 $u$ 为\textbf{调和函数}。
\end{dingliyi}

\begin{dingli}{解析函数的调和性质}{}
设
\[
    f(z)=u(x,y)+\mathrm{i}v(x,y)
\]
在区域 $D$ 内解析，并且 $u,v$ 在 $D$ 内有连续的二阶偏导数，则 $u$ 和 $v$ 都是调和函数。
\end{dingli}

\begin{proof}
由柯西--黎曼条件，
\[
    u_x=v_y,\qquad u_y=-v_x.
\]
对第一式再对 $x$ 求偏导，对第二式再对 $y$ 求偏导，得
\[
    u_{xx}=v_{yx},\qquad u_{yy}=-v_{xy}.
\]
由于二阶偏导连续，可交换求导次序，所以
\[
    v_{yx}=v_{xy}.
\]
于是
\[
    u_{xx}+u_{yy}=v_{xy}-v_{xy}=0.
\]
所以 $u$ 是调和函数。

同理，对
\[
    v_y=u_x,\qquad v_x=-u_y
\]
做同样处理，可得
\[
    v_{xx}+v_{yy}=0.
\]
所以 $v$ 也是调和函数。
\end{proof}

\begin{lizi}{验证实部和虚部都是调和函数}{}
对
\[
    f(z)=z^2=(x^2-y^2)+\mathrm{i}(2xy),
\]
验证其实部和虚部都满足拉普拉斯方程。

\begin{jie}
实部为
\[
    u=x^2-y^2,
\]
所以
\[
    u_{xx}=2,\qquad u_{yy}=-2,
\]
从而
\[
    u_{xx}+u_{yy}=0.
\]

虚部为
\[
    v=2xy,
\]
所以
\[
    v_{xx}=0,\qquad v_{yy}=0,
\]
从而
\[
    v_{xx}+v_{yy}=0.
\]
因此 $u$ 和 $v$ 都是调和函数。
\end{jie}
\end{lizi}

\subsection{共轭调和函数}

\begin{dingliyi}{共轭调和函数}{}
若两个调和函数 $u(x,y)$ 和 $v(x,y)$ 满足柯西--黎曼条件
\[
    u_x=v_y,\qquad u_y=-v_x,
\]
则称 $u$ 和 $v$ 互为\textbf{共轭调和函数}。
\end{dingliyi}

\begin{zhu}{}{}
如果 $u$ 和 $v$ 互为共轭调和函数，那么
\[
    f(z)=u+\mathrm{i}v
\]
就是一个解析函数。反过来，解析函数的实部和虚部一定互为共轭调和函数。
\end{zhu}

\subsection{由已知实部求共轭函数}

\begin{lizi}{由实部求共轭调和函数}{}
已知解析函数的实部为
\[
    u(x,y)=x^2-y^2,
\]
求一个对应的共轭调和函数 $v(x,y)$。

\begin{jie}
由柯西--黎曼条件，
\[
    v_y=u_x=2x,\qquad v_x=-u_y=2y.
\]
先由
\[
    v_y=2x
\]
对 $y$ 积分，得
\[
    v(x,y)=2xy+\varphi(x),
\]
其中 $\varphi(x)$ 只依赖于 $x$。

再对上式求 $x$ 导数，得
\[
    v_x=2y+\varphi'(x).
\]
而根据柯西--黎曼条件，必须有
\[
    v_x=2y.
\]
因此
\[
    \varphi'(x)=0,
\]
从而 $\varphi(x)$ 是常数。于是一个共轭调和函数为
\[
    v(x,y)=2xy+C.
\]
通常把常数并入函数中，写成
\[
    v(x,y)=2xy.
\]
\end{jie}
\end{lizi}

\begin{lizi}{判断一个调和函数能否作为解析函数的实部}{}
讨论
\[
    u(x,y)=x^2+y^2
\]
能否作为某个解析函数的实部。

\begin{jie}
若 $u$ 是某个解析函数的实部，那么它必须先是调和函数。因此先检查它是否满足拉普拉斯方程。

计算得
\[
    u_x=2x,\qquad u_y=2y,
\]
进一步有
\[
    u_{xx}=2,\qquad u_{yy}=2.
\]
于是
\[
    u_{xx}+u_{yy}=4\neq 0.
\]

所以 $u(x,y)=x^2+y^2$ 不是调和函数。既然连调和函数都不是，它就不可能作为某个解析函数的实部。

这个例子提醒我们：``给一个函数找共轭函数''并不是总能做到的。能否嵌入解析函数，首先要看它自己是否满足拉普拉斯方程。
\end{jie}
\end{lizi}

\subsection{什么时候能在整个区域里找到共轭函数}

刚才这两个例子容易让人形成一种印象：好像只要函数是调和的，我们总能把另一个共轭函数直接算出来。事情并没有这么简单。真正会影响``能不能在整个区域里拼起来''的，往往不是偏导会不会算，而是区域里有没有洞。

这也正是为什么第一节一开始就要提前讲``单连通''。当时它看起来像是在为复积分铺路；其实到了这里，它又会重新出现，决定一个调和函数能不能在整个区域里配出单值的共轭函数。

\begin{tishi}[一个先记住的事实]
在本书后面的使用层面，可以先记住下面这句话：若区域 $D$ 是单连通的，而调和函数 $u$ 又足够光滑，那么它可以在整个 $D$ 内找到共轭调和函数；而且这样的共轭函数除了加一个常数外是唯一的。

但若区域有洞，局部上虽然常常还能解出满足柯西--黎曼条件的函数，整体上却未必能选成单值函数。
\end{tishi}

\begin{lizi}{为什么有洞会带来麻烦}{}
在区域
\[
    D=\mathbb{C}\setminus\{0\}
\]
上考虑
\[
    u(x,y)=\ln|z|=\frac{1}{2}\ln(x^2+y^2).
\]
说明它虽然是调和函数，但在整个 $D$ 上没有单值的共轭调和函数。

\begin{jie}
先计算偏导数：
\[
    u_x=\frac{x}{x^2+y^2},\qquad u_y=\frac{y}{x^2+y^2}.
\]
于是若存在共轭调和函数 $v$，它就必须满足
\[
    v_y=u_x=\frac{x}{x^2+y^2},\qquad
    v_x=-u_y=-\frac{y}{x^2+y^2}.
\]

和这组偏导关系最匹配的候选函数，其实就是极角
\[
    v(x,y)=\arg z.
\]
因为在不绕过原点的一小块区域里，$\arg z$ 的偏导确实满足上面的关系。

但问题也正出在这里：当我们绕原点转一圈后，辐角会增加 $2\pi$。这说明在整个去心平面
\[
    \mathbb{C}\setminus\{0\}
\]
上，$\arg z$ 不能选成一个单值连续函数。

因此，$u=\ln|z|$ 虽然在 $D$ 内调和，却不能在整个 $D$ 上配出单值的共轭调和函数。也就是说，它只有局部共轭函数，没有全局单值的共轭函数。
\end{jie}
\end{lizi}

\begin{tishi}[这里和第一章、后面章节的联系]
这里真正卡住的，不是柯西--黎曼方程不会解，而是区域的拓扑结构和辐角的多值性在起作用。第一章里已经见过 $\arg z$ 的多值性；后面讨论对数函数、多值函数与分支时，这个例子还会再次出现。
\end{tishi}

\subsection{等值线的正交性}

\begin{dingli}{共轭调和函数等值线的正交性}{}
设 $u(x,y)$ 与 $v(x,y)$ 互为共轭调和函数。在满足
\[
    \nabla u\neq 0,\qquad \nabla v\neq 0
\]
的点附近，两族曲线
\[
    u(x,y)=C_1,\qquad v(x,y)=C_2
\]
彼此正交。
\end{dingli}

\begin{proof}
曲线
\[
    u(x,y)=C_1
\]
的法向量为
\[
    \nabla u=(u_x,u_y),
\]
而曲线
\[
    v(x,y)=C_2
\]
的法向量为
\[
    \nabla v=(v_x,v_y).
\]
由柯西--黎曼条件，
\[
    u_x=v_y,\qquad u_y=-v_x.
\]
于是
\[
    \nabla u\cdot\nabla v=u_xv_x+u_yv_y=v_yv_x+(-v_x)v_y=0.
\]
两条曲线的法向量垂直，所以它们本身也彼此正交。
\end{proof}

\section{二维势场与解析函数的局部几何意义}

\begin{wulibj}[为什么要用这一节收束全章]
如果第二章只停在定义、定理和偏导关系上，学生很容易觉得这一章只是复杂的符号操练。本节的作用，就是把前面得到的数学结构重新放回图像和物理问题里：解析函数的实部和虚部，可以看成同一个二维场的两张正交地图；而解析函数的导数，在局部上又像第一章里学过的``乘以一个复数''一样，对小图形施加统一的伸缩和旋转。

本节不是要把保角映射系统讲完，也不是要把二维势场的边值问题一次求尽。它的任务更像是一张全景图：让学生在学完整章以后，第一次真正看清解析函数为什么既有物理意义，又有几何意义。
\end{wulibj}

\subsection{二维无源场与复势的最初图像}

在二维静电场、稳态温度场等问题里，常会遇到一个势函数 $\phi(x,y)$，它满足拉普拉斯方程
\[
    \phi_{xx}+\phi_{yy}=0.
\]
如果还能找到另一个函数 $\psi(x,y)$，使得
\[
    \phi_x=\psi_y,\qquad \phi_y=-\psi_x,
\]
那么
\[
    W(z)=\phi(x,y)+\mathrm{i}\psi(x,y)
\]
就是一个解析函数。这个函数通常称为\textbf{复势}。

\begin{zhu}{}{}
这里的 $\phi$ 和 $\psi$ 可以有不同的物理解释。在静电问题里，$\phi$ 常对应电势，$\psi$ 对应与电场线相关的流函数；在稳态导热中，$\phi$ 可以对应温度分布，$\psi$ 则与热流线相关。具体名字会因问题而异，但数学结构是一样的。
\end{zhu}

\begin{lizi}{均匀场对应的最简单复势}{}
考察复函数
\[
    W(z)=z=x+\mathrm{i}y.
\]
说明它怎样描述一个最简单的二维均匀场。

\begin{jie}
把 $W(z)$ 写成实部和虚部：
\[
    W(z)=x+\mathrm{i}y.
\]
因此
\[
    \phi(x,y)=x,\qquad \psi(x,y)=y.
\]

先看等值线。

由
\[
    \phi(x,y)=x=C_1
\]
可知，$\phi$ 的等值线是一族竖直直线。

由
\[
    \psi(x,y)=y=C_2
\]
可知，$\psi$ 的等值线是一族水平直线。

它们显然彼此正交。这正符合前面讲过的解析函数实部与虚部的等值线正交这一结论。

再看物理意义。如果把 $\phi$ 看成某个势函数，那么它沿 $x$ 方向线性变化，沿 $y$ 方向不变，因此对应的是一个方向恒定、大小也恒定的二维均匀场。换句话说，这个例子描述的是最简单的一类无源场：所有场线彼此平行，所有等势线也彼此平行，并且这两族直线互相垂直。

这个例子虽然简单，却很重要，因为它告诉我们：解析函数的实部和虚部并不只是形式上的两个函数，它们常常就是同一个二维物理场的两张互相垂直的``地图''。
\end{jie}
\end{lizi}

\subsection{等势线与场线的正交}

\begin{dingli}{等势线与场线正交}{}
若
\[
    W(z)=\phi(x,y)+\mathrm{i}\psi(x,y)
\]
在某区域内解析，则两族曲线
\[
    \phi(x,y)=C_1,\qquad \psi(x,y)=C_2
\]
在正则点附近彼此正交。
\end{dingli}

\begin{proof}
这其实就是上一节正交性定理的直接物理解释。因为解析性保证 $\phi$ 与 $\psi$ 满足柯西--黎曼条件，所以它们的梯度点积为零，从而两族等值线正交。
\end{proof}

\subsection{导数的局部作用}

设 $f(z)$ 在点 $z_0$ 可导，则由导数定义，
\[
    \frac{f(z)-f(z_0)}{z-z_0}\to f'(z_0)\qquad (z\to z_0).
\]
把它改写一下，就得到
\begin{equation}\label{eq:ch02-local-linear}
    f(z)-f(z_0)=f'(z_0)(z-z_0)+o(|z-z_0|).
\end{equation}

这句话的意思非常重要：当 $z$ 足够接近 $z_0$ 时，映射 $w=f(z)$ 的主要作用，几乎就等于先把向量 $z-z_0$ 乘上一个复数 $f'(z_0)$，再平移到点 $f(z_0)$。

若把
\[
    f'(z_0)=\rho e^{\mathrm{i}\theta},\qquad \rho=|f'(z_0)|,
\]
写成极形式，那么乘以 $f'(z_0)$ 的几何意义就很清楚了：
\begin{itemize}
    \item 长度被放大或缩小为原来的 $\rho$ 倍；
    \item 方向统一旋转了角度 $\theta$。
\end{itemize}

\begin{tishi}[这里和第一章的联系]
第一章里，我们已经知道：乘以一个非零复数，几何上对应``伸缩加旋转''。式 \eqref{eq:ch02-local-linear} 告诉我们，解析函数在足够小的尺度上，也正是这样起作用的。也就是说，解析函数的导数把第一章的复数几何意义重新带了回来。
\end{tishi}

\subsection{局部保角性的预告}

\begin{dingli}{局部保角性的预告}{}
若函数 $f(z)$ 在点 $z_0$ 解析，且
\[
    f'(z_0)\neq 0,
\]
则它在 $z_0$ 附近局部保持角度。
\end{dingli}

\begin{proof}
由式 \eqref{eq:ch02-local-linear}，
\[
    f(z)-f(z_0)=f'(z_0)(z-z_0)+o(|z-z_0|).
\]
当 $z$ 足够靠近 $z_0$ 时，主导作用来自一次项 $f'(z_0)(z-z_0)$。而乘以非零复数 $f'(z_0)$ 只会对所有方向统一施加同一个旋转和伸缩，因此曲线间的夹角大小保持不变，方向也保持一致。所以 $f$ 在 $z_0$ 附近局部保角。
\end{proof}

\begin{jinshi}[为什么一定要要求 $f'(z_0)\neq 0$]
如果 $f'(z_0)=0$，局部主导项不再是一次项，这时映射可能会把角度压扁或多重覆盖。例如 $f(z)=z^2$ 在 $z=0$ 处就不能简单地看成一个保角映射。
\end{jinshi}

\section{本章小结}

\begin{tishi}[一句话回看本章]
这一章的核心任务，是把``复数的几何语言''真正升级成``解析函数的结构语言''：从区域和曲线出发，经由极限、连续、导数和柯西--黎曼条件，最终看见了解析函数与调和函数、二维势场和局部保角性的联系。
\end{tishi}

本章的主线可以整理成：
\[
\begin{aligned}
    \text{区域与曲线}
    &\Longrightarrow \text{复变函数}
    \Longrightarrow \text{极限与连续}
    \Longrightarrow \text{导数与解析}\\
    &\Longrightarrow \text{C-R 条件}
    \Longrightarrow \text{调和性质}
    \Longrightarrow \text{物理与几何图像}.
\end{aligned}
\]

\begin{tishi}[本章最重要的结论]
\begin{itemize}
    \item 复变极限必须对任意路径趋近都给出同一个结果。
    \item 复导数的存在条件远比实导数严格，因为增量 $\Delta z$ 可以从任意方向趋近于零。
    \item 柯西--黎曼条件
    \[
        u_x=v_y,\qquad u_y=-v_x
    \]
    是可导判别的核心工具。
    \item 解析函数的实部和虚部都是调和函数，并互为共轭调和函数。
    \item 当 $f'(z_0)\neq 0$ 时，解析函数在 $z_0$ 附近局部上像乘以一个非零复数，因此会局部保角。
\end{itemize}
\end{tishi}

\begin{tishi}[做题时怎样判断从哪里下手]
\begin{itemize}
    \item 看到极限是否存在的问题，先想``不同路径会不会给出不同结果''；若要证明极限不存在，优先找两条最简单的路径。
    \item 看到``是否可导''的问题，先把 $f(z)$ 写成 $u(x,y)+\mathrm{i}v(x,y)$，再判断是用导数定义直接算，还是用柯西--黎曼条件判别更省力。
    \item 看到柯西--黎曼条件成立时，不要立刻下结论；还要继续检查偏导是否连续，或者它是不是只在某一点上偶然成立。
    \item 已知实部或虚部要求解析函数时，先检查给出的函数是否调和，再用柯西--黎曼条件去求另一个部分，并记得共轭函数只差一个加法常数。
    \item 一旦区域里有洞，就要提高警惕：局部上能写出的共轭函数，未必能在整个区域里选成单值函数。
    \item 看到等势线、场线或局部角度这类几何和物理描述时，要主动想到``解析函数的实部虚部正交''以及``导数对应局部旋转加伸缩''。
\end{itemize}
\end{tishi}

\begin{jinshi}[本章最容易混淆的地方]
\begin{itemize}
    \item 不要把``在一点可导''和``解析''混为一谈。
    \item 不要把``满足柯西--黎曼条件''机械等同于``可导''；充分性还需要附加条件。
    \item 不要把保角理解成``长度和面积都保持不变''。保角的重点是角度保持，而不是距离保持。
\end{itemize}
\end{jinshi}

\begin{tishi}[和后面章节的联系]
\begin{itemize}
    \item 下一章的复积分，将直接使用本章建立的区域、曲线和单连通语言。
    \item 第四章的泰勒展开与洛朗展开，都建立在本章的``解析性''概念上。
    \item 后面的数理方程部分会再次遇到拉普拉斯方程，本章关于调和函数的结论正是一个重要连接点。
    \item 选学的保角映射章，则会把本章最后的``局部保角性''系统展开。
\end{itemize}
\end{tishi}

\section{习题}

\textbf{基础题}

\begin{enumerate}
    \item 判断下列集合是否为区域；若是，再判断其是否单连通：
    \[
        \{z:|z|<2\},\qquad \{z:1<|z|<3\},\qquad \mathbb{C}\setminus\{0\},\qquad \{z:\operatorname{Im}z>0\}.
    \]

    \item 把下列函数写成 $u(x,y)+\mathrm{i}v(x,y)$ 的形式：
    \[
        z^2,\qquad \frac{1}{z},\qquad \bar z.
    \]

    \item 讨论下列极限是否存在：
    \[
        \lim_{z\to 0}\frac{\bar z}{z},\qquad
        \lim_{z\to 0}|z|^2.
    \]

    \item 讨论函数
    \[
        f(z)=\frac{z+1}{z-\mathrm{i}}
    \]
    在哪些点连续。

    \item 用导数定义求下列函数的导数：
    \[
        f(z)=z^2,\qquad f(z)=z^3.
    \]
\end{enumerate}

\textbf{综合题}

\begin{enumerate}[resume]
    \item 用柯西--黎曼条件判断下列函数在何处可导，并在可导处求导数：
    \[
        f(z)=z^2,\qquad f(z)=\bar z,\qquad f(z)=|z|^2.
    \]

    \item 已知解析函数的实部为
    \[
        u(x,y)=x^2-y^2,
    \]
    求其一个共轭调和函数。

    \item 证明：若
    \[
        f(z)=u(x,y)+\mathrm{i}v(x,y)
    \]
    在区域内解析，且 $u,v$ 具有连续二阶偏导数，则 $u$ 和 $v$ 都满足拉普拉斯方程。

    \item 说明为什么
    \[
        u(x,y)=x^2+y^2
    \]
    不能作为某个解析函数的实部。
\end{enumerate}

\textbf{挑战题}

\begin{enumerate}[resume]
    \item 设
    \[
        f(z)=
        \begin{cases}
            \dfrac{x^3}{x^2+y^2}+\mathrm{i}\dfrac{y^3}{x^2+y^2}, & z\neq 0,\\[1em]
            0, & z=0,
        \end{cases}
    \]
    验证它在原点满足柯西--黎曼条件，但在原点不可导。

    \item 设 $f(z)$ 在 $z_0$ 处解析且 $f'(z_0)\neq 0$。结合本章最后的局部线性化公式，说明为什么它在 $z_0$ 附近局部保持角度。

    \item 研究映射
    \[
        w=z^2
    \]
    在 $z_0\neq 0$ 与 $z_0=0$ 附近的局部几何作用有何不同，并解释这和条件 $f'(z_0)\neq 0$ 的关系。

    \item 设
    \[
        u(x,y)=\ln\sqrt{x^2+y^2}=\ln|z|,\qquad D=\mathbb{C}\setminus\{0\}.
    \]
    验证 $u$ 在 $D$ 内调和，并说明为什么它在整个 $D$ 上不能取单值的共轭调和函数。
\end{enumerate}
